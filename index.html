#include "BluetoothSerial.h" #include BluetoothSerial SerialBT; // الأقطاب السليمة التي تم اختبارها بنجاح للتقدم للأمام const int LEFT_MOTOR_PIN = 26; // المحرك الأيسر const int RIGHT_MOTOR_PIN = 14; // المحرك الأيمن // 5-channel line sensor const int SENSOR1 = 32; // far left const int SENSOR2 = 33; const int SENSOR3 = 34; // center const int SENSOR4 = 35; const int SENSOR5 = 39; // far right // لون الخط (أسود = LOW) const int BLACK_LINE_STATE = LOW; enum RobotState { WAITING_FOR_1, FOLLOWING_PHASE_1, PAUSED_PHASE_1, WAITING_FOR_3, FOLLOWING_PHASE_2, PAUSED_PHASE_2, MANUAL_CONTROL }; RobotState currentState = WAITING_FOR_1; // سرعات متوافقة مع قدرات المواتير (PWM) int baseStraight = 115; int baseCurve = 110; int baseSharp = 100; int searchSpeed = 105; // عوامل التصحيح (PD Controller) float KP = 0.30; float KD = 0.08; const int MAX_CORRECTION = 55; const int MIN_EFFECTIVE_PWM = 100; // منعاً للتوقف (Stall) const unsigned long FINISH_IGNORE_AFTER_START_MS = 1500; // تجاهل خط النهاية في أول ثانية ونصف const unsigned long LOST_BEFORE_SEARCH_MS = 150; unsigned long lineFollowStartedAt = 0; unsigned long lineLostAt = 0; unsigned long pauseStartedAt = 0; int lastError = 0; int lastLineDirection = 0; int lastTurnDirection = 0; // متغيرات الوضع اليدوي int manualSpeed = 113; // تعادل السرعة 4 char manualCommand = 'S'; bool debugSensors = false; unsigned long lastDebugAt = 0; #if ESP_ARDUINO_VERSION_MAJOR >= 3 const int CH_L_MOTOR = 0; const int CH_R_MOTOR = 1; void setupPwmPin(int pin, int channel) { ledcAttach(pin, 5000, 8); } void writePwm(int pin, int channel, int value) { value = constrain(value, 0, 255); ledcWrite(pin, value); } #else const int PWM_FREQ = 5000; const int PWM_RESOLUTION = 8; const int CH_L_MOTOR = 0; const int CH_R_MOTOR = 1; void setupPwmPin(int pin, int channel) { ledcSetup(channel, PWM_FREQ, PWM_RESOLUTION); ledcAttachPin(pin, channel); } void writePwm(int pin, int channel, int value) { value = constrain(value, 0, 255); ledcWrite(channel, value); } #endif typedef struct { int s1; int s2; int s3; int s4; int s5; int blackCount; int error; bool anyLine; } SensorRead; void setupPwmPin(int pin, int channel); void writePwm(int pin, int channel, int value); void handleBluetooth(); void resetTracking(); SensorRead readSensors(); int stableDigitalRead(int pin); void addSensorWeight(int sensorValue, int weight, int &weightedSum, int &blackCount); void followLine(SensorRead sensors); int cleanMotorSpeed(int speed); void recoverLostLine(); void searchForLine(); void printSensorDebug(SensorRead sensors); void moveForward(int speedLeft, int speedRight); void turnLeft(int speed); void turnRight(int speed); void stopRobot(); void executeManualMovement(); void setup() { Serial.begin(115200); SerialBT.begin("ESP32_LineFollower"); Serial.println("Ready. Waiting for '1' to start, or 'n' for Manual Mode."); setupPwmPin(LEFT_MOTOR_PIN, CH_L_MOTOR); setupPwmPin(RIGHT_MOTOR_PIN, CH_R_MOTOR); pinMode(SENSOR1, INPUT); pinMode(SENSOR2, INPUT); pinMode(SENSOR3, INPUT); pinMode(SENSOR4, INPUT); pinMode(SENSOR5, INPUT); stopRobot(); } void loop() { handleBluetooth(); SensorRead sensors = readSensors(); printSensorDebug(sensors); switch (currentState) { case WAITING_FOR_1: case WAITING_FOR_3: case MANUAL_CONTROL: // يتم التحكم في هذه الحالات عبر البلوتوث break; case FOLLOWING_PHASE_1: // إذا قرأت الحساسات الخمسة خط النهاية if (sensors.blackCount == 5 && millis() - lineFollowStartedAt >= FINISH_IGNORE_AFTER_START_MS) { stopRobot(); pauseStartedAt = millis(); currentState = PAUSED_PHASE_1; Serial.println("Phase 1 Finished. Pausing for 3 seconds..."); } else { followLine(sensors); } break; case PAUSED_PHASE_1: stopRobot(); if (millis() - pauseStartedAt >= 3000) { SerialBT.println("2"); currentState = WAITING_FOR_3; Serial.println("Sent 2. Waiting for 3..."); } break; case FOLLOWING_PHASE_2: // إذا قرأت الحساسات الخمسة خط النهاية مرة أخرى if (sensors.blackCount == 5 && millis() - lineFollowStartedAt >= FINISH_IGNORE_AFTER_START_MS) { stopRobot(); pauseStartedAt = millis(); currentState = PAUSED_PHASE_2; Serial.println("Phase 2 Finished. Pausing for 3 seconds..."); } else { followLine(sensors); } break; case PAUSED_PHASE_2: stopRobot(); if (millis() - pauseStartedAt >= 3000) { SerialBT.println("0"); currentState = WAITING_FOR_1; // العودة لنقطة البداية الأساسية Serial.println("Sent 0. Waiting for 1 to restart cycle..."); } break; } delay(5); } void handleBluetooth() { if (!SerialBT.available()) return; char c = SerialBT.read(); // التحويل بين الأوضاع if (c == 'n') { currentState = MANUAL_CONTROL; manualCommand = 'S'; stopRobot(); Serial.println("Switched to MANUAL MODE"); return; } else if (c == 'N') { currentState = WAITING_FOR_1; stopRobot(); Serial.println("Switched to AUTO MODE (Waiting for 1)"); return; } // أوامر الوضع اليدوي if (currentState == MANUAL_CONTROL) { c = toupper(c); if (c == 'F' || c == 'L' || c == 'R' || c == 'S') { manualCommand = c; executeManualMovement(); } else if (c >= '0' && c <= '9') { manualSpeed = map(c - '0', 0, 9, 0, 255); executeManualMovement(); } } // أوامر الوضع الآلي else { if (c == '1' && (currentState == WAITING_FOR_1)) { resetTracking(); currentState = FOLLOWING_PHASE_1; Serial.println("Starting Phase 1..."); } else if (c == '3' && currentState == WAITING_FOR_3) { resetTracking(); currentState = FOLLOWING_PHASE_2; Serial.println("Starting Phase 2..."); } else if (c == 'S' || c == 's') { currentState = WAITING_FOR_1; stopRobot(); Serial.println("Emergency Stop. Resetting cycle."); } else if (c == 'D' || c == 'd') { debugSensors = !debugSensors; } } } void executeManualMovement() { switch (manualCommand) { case 'F': moveForward(manualSpeed, manualSpeed); break; case 'L': turnLeft(manualSpeed); break; case 'R': turnRight(manualSpeed); break; case 'S': stopRobot(); break; } } void resetTracking() { lineLostAt = 0; lastError = 0; lastLineDirection = 0; lastTurnDirection = 0; lineFollowStartedAt = millis(); } SensorRead readSensors() { SensorRead r; r.s1 = stableDigitalRead(SENSOR1); r.s2 = stableDigitalRead(SENSOR2); r.s3 = stableDigitalRead(SENSOR3); r.s4 = stableDigitalRead(SENSOR4); r.s5 = stableDigitalRead(SENSOR5); r.blackCount = 0; int weightedSum = 0; addSensorWeight(r.s1, -200, weightedSum, r.blackCount); addSensorWeight(r.s2, -100, weightedSum, r.blackCount); addSensorWeight(r.s3, 0, weightedSum, r.blackCount); addSensorWeight(r.s4, 100, weightedSum, r.blackCount); addSensorWeight(r.s5, 200, weightedSum, r.blackCount); r.anyLine = r.blackCount > 0; r.error = r.anyLine ? weightedSum / r.blackCount : lastError; return r; } int stableDigitalRead(int pin) { int lowCount = 0; int highCount = 0; for (int i = 0; i < 3; i++) { int value = digitalRead(pin); if (value == LOW) lowCount++; else highCount++; delayMicroseconds(180); } return lowCount > highCount ? LOW : HIGH; } void addSensorWeight(int sensorValue, int weight, int &weightedSum, int &blackCount) { if (sensorValue == BLACK_LINE_STATE) { weightedSum += weight; blackCount++; } } void followLine(SensorRead sensors) { if (!sensors.anyLine) { recoverLostLine(); return; } lineLostAt = 0; lastLineDirection = sensors.error < -35 ? -1 : (sensors.error > 35 ? 1 : 0); int absError = abs(sensors.error); int baseSpeed = baseStraight; if (absError > 130 || sensors.blackCount >= 3) { baseSpeed = baseSharp; } else if (absError > 45 || sensors.blackCount >= 2) { baseSpeed = baseCurve; } float rawCorrection = (KP * sensors.error) + (KD * (sensors.error - lastError)); int correction = constrain((int)rawCorrection, -MAX_CORRECTION, MAX_CORRECTION); int leftSpeed = baseSpeed + correction; int rightSpeed = baseSpeed - correction; if (correction > 8) lastTurnDirection = 1; else if (correction < -8) lastTurnDirection = -1; moveForward(cleanMotorSpeed(leftSpeed), cleanMotorSpeed(rightSpeed)); lastError = sensors.error; } int cleanMotorSpeed(int speed) { speed = constrain(speed, 0, 150); if (speed > 0 && speed < MIN_EFFECTIVE_PWM) { speed = MIN_EFFECTIVE_PWM; // الحفاظ على دوران الموتور لتجنب الطنين } return speed; } void recoverLostLine() { if (lineLostAt == 0) lineLostAt = millis(); unsigned long lostTime = millis() - lineLostAt; if (lostTime < LOST_BEFORE_SEARCH_MS) { if (lastTurnDirection > 0) moveForward(100, 0); else if (lastTurnDirection < 0) moveForward(0, 100); else moveForward(100, 100); return; } searchForLine(); } void searchForLine() { if (lastTurnDirection < 0) turnLeft(searchSpeed); else if (lastTurnDirection > 0) turnRight(searchSpeed); else if (lastLineDirection < 0 || lastError < 0) turnLeft(searchSpeed); else turnRight(searchSpeed); } void printSensorDebug(SensorRead sensors) { if (!debugSensors || millis() - lastDebugAt < 180) return; lastDebugAt = millis(); SerialBT.print("S:"); SerialBT.print(sensors.s1); SerialBT.print(" "); SerialBT.print(sensors.s2); SerialBT.print(" "); SerialBT.print(sensors.s3); SerialBT.print(" "); SerialBT.print(sensors.s4); SerialBT.print(" "); SerialBT.print(sensors.s5); SerialBT.print(" Blk="); SerialBT.print(sensors.blackCount); SerialBT.print(" Err="); SerialBT.println(sensors.error); } // دوال الحركة الأساسية المعاد تصميمها لتناسب الهاردوير الحالي void moveForward(int speedLeft, int speedRight) { writePwm(LEFT_MOTOR_PIN, CH_L_MOTOR, speedLeft); writePwm(RIGHT_MOTOR_PIN, CH_R_MOTOR, speedRight); } void turnLeft(int speed) { writePwm(LEFT_MOTOR_PIN, CH_L_MOTOR, 0); // الأيسر يتوقف writePwm(RIGHT_MOTOR_PIN, CH_R_MOTOR, speed); // الأيمن يدور } void turnRight(int speed) { writePwm(LEFT_MOTOR_PIN, CH_L_MOTOR, speed); // الأيسر يدور writePwm(RIGHT_MOTOR_PIN, CH_R_MOTOR, 0); // الأيمن يتوقف } void stopRobot() { writePwm(LEFT_MOTOR_PIN, CH_L_MOTOR, 0); writePwm(RIGHT_MOTOR_PIN, CH_R_MOTOR, 0); }
